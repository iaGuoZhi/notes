Practical Examples

# Typical Linux boot sequence
set root=(hd0,gpt2)
linux /vmlinuz root=/dev/sda2 ro
initrd /initramfs.img
boot

# Boot from ISO file
loopback loop /path/to/image.iso
linux (loop)/casper/vmlinuz boot=casper iso-scan/filename=/image.iso
initrd (loop)/casper/initrd
boot
